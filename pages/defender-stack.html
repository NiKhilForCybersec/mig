Cross-Product Threat Hunting

Advanced hunting query correlating signals across MDE, MDI, MDO, and Sentinel.

// Cross-Product Attack Investigation
// Correlate email → endpoint → identity signals

let suspiciousUser = "user@company.com";
let timeWindow = 24h;

// Stage 1: Check for phishing emails received
let phishingEmails = EmailEvents
    | where TimeGenerated > ago(timeWindow)
    | where RecipientEmailAddress == suspiciousUser
    | where ThreatTypes has "Phish" or DeliveryAction == "Delivered"
    | project EmailTime = TimeGenerated, Subject, SenderFromAddress, NetworkMessageId;

// Stage 2: Check for suspicious file downloads
let fileDownloads = DeviceFileEvents
    | where TimeGenerated > ago(timeWindow)
    | where InitiatingProcessAccountUpn == suspiciousUser
    | where ActionType == "FileCreated"
    | where FileName endswith ".exe" or FileName endswith ".dll"
    | project FileTime = TimeGenerated, FileName, FolderPath, SHA256;

// Stage 3: Check for process execution
let processExec = DeviceProcessEvents
    | where TimeGenerated > ago(timeWindow)
    | where InitiatingProcessAccountUpn == suspiciousUser
    | where FileName in~ ("powershell.exe", "cmd.exe", "wscript.exe")
    | project ProcTime = TimeGenerated, FileName, ProcessCommandLine;

// Stage 4: Check identity anomalies
let identityRisk = SigninLogs
    | where TimeGenerated > ago(timeWindow)
    | where UserPrincipalName == suspiciousUser
    | where RiskLevelDuringSignIn in ("medium", "high")
    | project SigninTime = TimeGenerated, IPAddress, RiskLevelDuringSignIn, Location;

// Combine all signals into timeline
phishingEmails
| join kind=fullouter (fileDownloads) on $left.EmailTime == $right.FileTime
| join kind=fullouter (processExec) on $left.EmailTime == $right.ProcTime
| join kind=fullouter (identityRisk) on $left.EmailTime == $right.SigninTime
| project
    InvestigatedUser = suspiciousUser,
    PhishingEmail = Subject,
    Sender = SenderFromAddress,
    SuspiciousFile = FileName,
    ProcessExecuted = FileName1,
    CommandLine = ProcessCommandLine,
    RiskySignIn = RiskLevelDuringSignIn,
    SignInIP = IPAddress
