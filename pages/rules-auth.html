1.1 Brute Force Login Attempts

High

Detects 10+ failed login attempts from the same source IP within 5 minutes

T1110 
-- Rule: Brute_Force_Login_Attempts
-- Trigger: 10+ failed logins from same source in 5 minutes

@RSA_Rule "Brute_Force_Detection"
SELECT * FROM Event
WHERE 
    event_category = 'authentication' AND
    event_outcome = 'failure' AND
    device_type IN ('windows', 'linux', 'activedirectory')
GROUP BY source_ip, destination_user
HAVING COUNT(*) >= 10
WINDOW SLIDING 5 MINUTES
ALERT severity = 'high', 
      module = 'authentication',
      technique = 'T1110'
// Rule: Brute Force Login Attempts
// MITRE: T1110 - Brute Force

let threshold = 10;
let timeWindow = 5m;

SigninLogs
| where TimeGenerated > ago(timeWindow)
| where ResultType != 0  // Failed logins
| summarize 
    FailedAttempts = count(),
    TargetAccounts = make_set(UserPrincipalName),
    FirstAttempt = min(TimeGenerated),
    LastAttempt = max(TimeGenerated)
    by IPAddress, Location
| where FailedAttempts >= threshold
| extend 
    AccountsTargeted = array_length(TargetAccounts),
    AttackDuration = LastAttempt - FirstAttempt
| project
    TimeGenerated = LastAttempt,
    IPAddress,
    Location,
    FailedAttempts,
    AccountsTargeted,
    TargetAccounts,
    AttackDuration,
    Severity = case(
        FailedAttempts >= 50, "High",
        FailedAttempts >= 25, "Medium",
        "Low"
    ),
    MITRETechnique = "T1110"

1.2 Impossible Travel Detection

Critical

Detects authentication from geographically distant locations within impossible timeframes

T1078 
-- Rule: Impossible_Travel_Detection
-- Trigger: Same user authenticates from distant locations

@RSA_Rule "Impossible_Travel"
SELECT 
    user_name,
    source_geo_country AS country1,
    LAST(source_geo_country) AS country2,
    source_ip AS ip1,
    LAST(source_ip) AS ip2,
    timestamp AS time1,
    LAST(timestamp) AS time2
FROM Event
WHERE 
    event_category = 'authentication' AND
    event_outcome = 'success'
GROUP BY user_name
HAVING 
    country1 != country2 AND
    TIMEDIFF(time2, time1) < 120 MINUTES AND
    GEO_DISTANCE(country1, country2) > 500
WINDOW SLIDING 2 HOURS
ALERT severity = 'critical',
      module = 'identity',
      technique = 'T1078'
// Rule: Impossible Travel Detection
// MITRE: T1078 - Valid Accounts

let timeWindow = 2h;
let impossibleTravelThresholdKm = 500;
let impossibleTravelTimeMinutes = 120;

SigninLogs
| where TimeGenerated > ago(timeWindow)
| where ResultType == 0  // Successful logins
| project 
    TimeGenerated, 
    UserPrincipalName, 
    IPAddress, 
    Location,
    Latitude = toreal(LocationDetails.geoCoordinates.latitude),
    Longitude = toreal(LocationDetails.geoCoordinates.longitude),
    City = tostring(LocationDetails.city),
    Country = tostring(LocationDetails.countryOrRegion)
| order by UserPrincipalName, TimeGenerated asc
| serialize
| extend 
    PrevLatitude = prev(Latitude, 1),
    PrevLongitude = prev(Longitude, 1),
    PrevTime = prev(TimeGenerated, 1),
    PrevLocation = prev(Location, 1),
    PrevIP = prev(IPAddress, 1),
    PrevUser = prev(UserPrincipalName, 1)
| where UserPrincipalName == PrevUser
| extend 
    TimeDiffMinutes = datetime_diff('minute', TimeGenerated, PrevTime),
    DistanceKm = geo_distance_2points(Longitude, Latitude, PrevLongitude, PrevLatitude) / 1000
| where DistanceKm > impossibleTravelThresholdKm
| where TimeDiffMinutes < impossibleTravelTimeMinutes
| extend 
    RequiredSpeedKmh = DistanceKm / (TimeDiffMinutes / 60.0)
| where RequiredSpeedKmh > 800  // Faster than commercial flight
| project
    TimeGenerated,
    UserPrincipalName,
    CurrentIP = IPAddress,
    CurrentLocation = Location,
    PreviousIP = PrevIP,
    PreviousLocation = PrevLocation,
    DistanceKm = round(DistanceKm, 2),
    TimeDiffMinutes,
    RequiredSpeedKmh = round(RequiredSpeedKmh, 2),
    Severity = "High",
    MITRETechnique = "T1078"

1.3 Service Account Interactive Logon

High

Detects service accounts used for interactive login sessions

T1078.002 
-- Rule: Service_Account_Interactive_Logon
-- Trigger: Service account used for interactive login

@RSA_Rule "Service_Account_Abuse"
SELECT * FROM Event
WHERE 
    event_category = 'authentication' AND
    event_outcome = 'success' AND
    logon_type IN (2, 10, 11) AND  -- Interactive, RemoteInteractive, CachedInteractive
    user_name MATCHES 'svc_*|srv_*|service_*|_svc$'
ALERT severity = 'high',
      module = 'identity',
      technique = 'T1078.002'
// Rule: Service Account Interactive Logon
// MITRE: T1078.002 - Domain Accounts

let serviceAccountPatterns = dynamic(['svc_', 'srv_', 'service_', '_svc', '_service']);

SecurityEvent
| where EventID == 4624  // Successful logon
| where LogonType in (2, 10, 11)  // Interactive logons
| where AccountType == "User"
| extend IsServiceAccount = array_length(
    array_filter(serviceAccountPatterns, p => Account contains p)
) > 0
| where IsServiceAccount == true
| project
    TimeGenerated,
    Account,
    Computer,
    LogonType,
    LogonTypeName = case(
        LogonType == 2, "Interactive",
        LogonType == 10, "RemoteInteractive",
        LogonType == 11, "CachedInteractive",
        "Unknown"
    ),
    IpAddress,
    WorkstationName,
    Severity = "High",
    MITRETechnique = "T1078.002"

1.4 After-Hours VPN Access

Medium

Detects VPN access outside business hours for non-exempt users

T1133 
-- Rule: After_Hours_VPN_Access
-- Trigger: VPN access outside business hours

@RSA_Rule "After_Hours_Access"
SELECT * FROM Event
WHERE 
    event_category = 'vpn' AND
    event_outcome = 'success' AND
    HOUR(timestamp) NOT BETWEEN 6 AND 20 AND
    DAYOFWEEK(timestamp) IN (1, 7) OR
    user_department NOT IN ('IT', 'Security', 'Executive')
ALERT severity = 'medium',
      module = 'identity',
      technique = 'T1133'
// Rule: After Hours VPN Access
// MITRE: T1133 - External Remote Services

let businessHoursStart = 6;
let businessHoursEnd = 20;
let exemptDepartments = dynamic(['IT', 'Security', 'Executive', 'Operations']);

SigninLogs
| where AppDisplayName contains "VPN" or 
        ResourceDisplayName contains "VPN" or
        ClientAppUsed == "VPN"
| where ResultType == 0
| extend 
    HourOfDay = hourofday(TimeGenerated),
    DayOfWeek = dayofweek(TimeGenerated),
    IsWeekend = dayofweek(TimeGenerated) in (0d, 6d)
| extend 
    IsAfterHours = HourOfDay < businessHoursStart or HourOfDay > businessHoursEnd
| where IsAfterHours or IsWeekend
| join kind=leftouter (
    IdentityInfo
    | summarize arg_max(TimeGenerated, *) by AccountUPN
    | project AccountUPN, Department, JobTitle
) on $left.UserPrincipalName == $right.AccountUPN
| where Department !in (exemptDepartments)
| project
    TimeGenerated,
    UserPrincipalName,
    Department,
    IPAddress,
    Location,
    HourOfDay,
    IsWeekend,
    DeviceDetail,
    Severity = "Medium",
    MITRETechnique = "T1133"

1.5 MFA Fatigue Attack Detection

Critical

Detects multiple MFA push notifications in short period (push fatigue attack)

T1621 
-- Rule: MFA_Fatigue_Attack
-- Trigger: Multiple MFA prompts in short period

@RSA_Rule "MFA_Fatigue"
SELECT * FROM Event
WHERE 
    event_category = 'authentication' AND
    event_type = 'mfa_challenge' AND
    event_outcome = 'pending'
GROUP BY user_name
HAVING COUNT(*) >= 5
WINDOW SLIDING 10 MINUTES
ALERT severity = 'critical',
      module = 'identity',
      technique = 'T1621'
// Rule: MFA Fatigue Attack Detection
// MITRE: T1621 - Multi-Factor Authentication Request Generation

let mfaThreshold = 5;
let timeWindow = 10m;

SigninLogs
| where TimeGenerated > ago(timeWindow)
| where ResultType == 50074  // MFA required
    or ResultType == 50076   // MFA required (strong auth)
    or ResultType == 50079   // MFA registration required
| summarize
    MFAPrompts = count(),
    Locations = make_set(Location),
    IPs = make_set(IPAddress),
    Devices = make_set(DeviceDetail.deviceId),
    FirstPrompt = min(TimeGenerated),
    LastPrompt = max(TimeGenerated)
    by UserPrincipalName
| where MFAPrompts >= mfaThreshold
| extend 
    UniqueLocations = array_length(Locations),
    UniqueIPs = array_length(IPs),
    AttackDuration = LastPrompt - FirstPrompt
| project
    TimeGenerated = LastPrompt,
    UserPrincipalName,
    MFAPrompts,
    UniqueLocations,
    UniqueIPs,
    AttackDuration,
    Severity = "Critical",
    MITRETechnique = "T1621",
    RecommendedAction = "Contact user immediately, consider disabling account"
