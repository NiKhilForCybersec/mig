7.1 Data Hoarding Detection

High

Detects unusual file access patterns indicating data staging for exfiltration

T1074 
-- Rule: Data_Hoarding_Detection
@RSA_Rule "Data_Hoarding"
SELECT * FROM Event
WHERE 
    event_type = 'file_access' AND
    action = 'copy'
GROUP BY user_name
HAVING COUNT(DISTINCT file_path) > 100
WINDOW SLIDING 24 HOURS
ALERT severity = 'high',
      technique = 'T1074'
// Rule: Data Hoarding Detection
// MITRE: T1074 - Data Staged

let fileThreshold = 100;
let lookback = 24h;

// Get baseline for normal user activity
let baseline = DeviceFileEvents
    | where TimeGenerated between (ago(14d) .. ago(1d))
    | where ActionType in ("FileCreated", "FileCopied", "FileModified")
    | summarize AvgDailyFiles = count() / 13 by InitiatingProcessAccountName
    | where AvgDailyFiles > 0;

DeviceFileEvents
| where TimeGenerated > ago(lookback)
| where ActionType in ("FileCreated", "FileCopied", "FileModified")
| summarize
    UniqueFiles = dcount(FileName),
    UniqueFolders = dcount(FolderPath),
    TotalBytes = sum(FileSize),
    FileTypes = make_set(tostring(split(FileName, ".")[-1]), 10),
    SampleFiles = make_set(FileName, 5)
    by InitiatingProcessAccountName, DeviceName
| where UniqueFiles > fileThreshold
| join kind=leftouter (baseline) on InitiatingProcessAccountName
| extend 
    DeviationFactor = iff(AvgDailyFiles > 0, UniqueFiles / AvgDailyFiles, 999),
    DataSizeMB = round(TotalBytes / (1024.0 * 1024), 2)
| where DeviationFactor > 3 // 3x normal activity
| project
    TimeGenerated = now(),
    UserName = InitiatingProcessAccountName,
    DeviceName,
    UniqueFilesAccessed = UniqueFiles,
    UniqueFolders,
    DataSizeMB,
    DeviationFactor = round(DeviationFactor, 2),
    FileTypes,
    SampleFiles,
    Severity = case(
        DeviationFactor > 10, "Critical",
        DeviationFactor > 5, "High",
        "Medium"
    ),
    MITRETechnique = "T1074"

7.2 Suspicious Email Forwarding

High

Detects email forwarding rules to external addresses

T1114.003 
-- Rule: Suspicious_Email_Forwarding
@RSA_Rule "Email_Forward_External"
SELECT * FROM Event
WHERE 
    event_source = 'exchange' AND
    operation = 'New-InboxRule' AND
    (forward_to NOT LIKE '*@company.com' OR
     redirect_to NOT LIKE '*@company.com')
ALERT severity = 'high',
      technique = 'T1114.003'
// Rule: Suspicious Email Forwarding Detection
// MITRE: T1114.003 - Email Collection: Email Forwarding Rule

let corporateDomains = dynamic(["@company.com", "@subsidiary.com"]);

OfficeActivity
| where TimeGenerated > ago(1d)
| where Operation in ("New-InboxRule", "Set-InboxRule", "Set-Mailbox")
| extend Parameters = parse_json(Parameters)
| mv-expand Parameters
| where Parameters.Name in ("ForwardTo", "ForwardAsAttachmentTo", "RedirectTo", "ForwardingSmtpAddress")
| extend ForwardAddress = tostring(Parameters.Value)
| where isnotempty(ForwardAddress)
| where not(ForwardAddress has_any (corporateDomains))
| project
    TimeGenerated,
    UserId,
    Operation,
    ForwardAddress,
    ClientIP,
    OriginatingServer,
    Severity = "High",
    MITRETechnique = "T1114.003",
    RecommendedAction = "Remove forwarding rule, investigate user activity"

7.3 Departing Employee Activity

Critical

Monitors employees on termination list for data exfiltration indicators

T1078 
-- Rule: Departing_Employee_Risk
@RSA_Rule "Departing_Employee"
SELECT * FROM Event
WHERE 
    user_name IN (SELECT user FROM hr.termination_list) AND
    (event_type = 'file_copy' OR
     event_type = 'usb_access' OR
     bytes_transferred > 10485760)
ALERT severity = 'critical',
      technique = 'T1078'
// Rule: Departing Employee Risk Monitoring
// MITRE: T1078 - Valid Accounts
// Requires: HR_TerminationList watchlist with columns: UserPrincipalName, TerminationDate

let departingUsers = _GetWatchlist('HR_TerminationList')
    | project UPN = tolower(UserPrincipalName), TermDate = TerminationDate;

let fileActivity = DeviceFileEvents
    | where TimeGenerated > ago(1d)
    | extend UPN = tolower(InitiatingProcessAccountUpn)
    | join kind=inner (departingUsers) on UPN
    | summarize 
        FilesCopied = countif(ActionType == "FileCopied"),
        FilesCreated = countif(ActionType == "FileCreated"),
        TotalBytes = sum(FileSize)
        by UPN, TermDate;

let usbActivity = DeviceEvents
    | where TimeGenerated > ago(1d)
    | where ActionType == "UsbDriveMounted"
    | extend UPN = tolower(InitiatingProcessAccountUpn)
    | join kind=inner (departingUsers) on UPN
    | summarize USBConnections = count() by UPN;

fileActivity
| join kind=leftouter (usbActivity) on UPN
| extend 
    USBConnections = coalesce(USBConnections, 0),
    DataSizeMB = round(TotalBytes / (1024.0 * 1024), 2)
| extend RiskScore = 
    (toint(FilesCopied > 50) * 30) +
    (toint(USBConnections > 0) * 40) +
    (toint(DataSizeMB > 100) * 30)
| where RiskScore > 0
| project
    TimeGenerated = now(),
    UserPrincipalName = UPN,
    TerminationDate = TermDate,
    FilesCopied,
    FilesCreated,
    USBConnections,
    DataSizeMB,
    RiskScore,
    Severity = case(
        RiskScore >= 70, "Critical",
        RiskScore >= 40, "High",
        "Medium"
    ),
    MITRETechnique = "T1078"
