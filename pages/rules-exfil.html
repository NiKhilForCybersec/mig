Data Exfiltration

Critical

32 RSA rules migrated to 30 Sentinel analytics rules for detecting large data transfers, cloud storage uploads, and email-based exfiltration.

MITRE Techniques: T1048 T1567 T1114

3.1 Large Outbound Data Transfer

High

Detects unusual large data transfers to external destinations

T1048 
-- Rule: Large_Outbound_Transfer
-- Trigger: Large data transfer to external IP

@RSA_Rule "Large_Data_Transfer"
SELECT * FROM Event
WHERE 
    direction = 'outbound' AND
    bytes_out > 104857600 -- 100MB AND
    destination_ip NOT IN (internal_ranges)
GROUP BY source_ip, destination_ip
WINDOW SLIDING 1 HOUR
ALERT severity = 'high',
      technique = 'T1048'
// Rule: Large Outbound Data Transfer
// MITRE: T1048 - Exfiltration Over Alternative Protocol

let thresholdMB = 100;
let thresholdBytes = thresholdMB * 1024 * 1024;

// Get baseline for user/device
let baseline = DeviceNetworkEvents
    | where TimeGenerated between (ago(7d) .. ago(1d))
    | where RemoteIPType == "Public"
    | summarize AvgDailyBytes = avg(SentBytes) by DeviceName;

DeviceNetworkEvents
| where TimeGenerated > ago(1h)
| where RemoteIPType == "Public"
| where SentBytes > 0
| summarize
    TotalBytesSent = sum(SentBytes),
    UniqueDestinations = dcount(RemoteIP),
    DestinationIPs = make_set(RemoteIP, 10),
    DestinationPorts = make_set(RemotePort, 10),
    ConnectionCount = count()
    by DeviceName, InitiatingProcessAccountName
| where TotalBytesSent > thresholdBytes
| join kind=leftouter (baseline) on DeviceName
| extend 
    BytesMB = round(TotalBytesSent / (1024.0 * 1024), 2),
    BaselineMB = round(coalesce(AvgDailyBytes, 0) / (1024.0 * 1024), 2),
    DeviationFactor = iff(AvgDailyBytes > 0, TotalBytesSent / AvgDailyBytes, 999)
| extend Severity = case(
    DeviationFactor > 10 or BytesMB > 500, "Critical",
    DeviationFactor > 5 or BytesMB > 250, "High",
    "Medium"
)
| project
    TimeGenerated = now(),
    DeviceName,
    UserName = InitiatingProcessAccountName,
    BytesSentMB = BytesMB,
    BaselineAvgMB = BaselineMB,
    DeviationFactor = round(DeviationFactor, 2),
    UniqueDestinations,
    DestinationIPs,
    Severity,
    MITRETechnique = "T1048"

3.2 Cloud Storage Upload Detection

High

Detects uploads to personal cloud storage services

T1567.002 
-- Rule: Cloud_Storage_Upload
-- Trigger: Upload to personal cloud storage

@RSA_Rule "Cloud_Upload"
SELECT * FROM Event
WHERE 
    http_method = 'POST' AND
    destination_host MATCHES '*dropbox*|*drive.google*|*onedrive*|*box.com*' AND
    bytes_out > 10485760 -- 10MB
ALERT severity = 'high',
      technique = 'T1567.002'
// Rule: Cloud Storage Upload Detection
// MITRE: T1567.002 - Exfiltration to Cloud Storage

let cloudStorageDomains = dynamic([
    "dropbox.com", "dropboxusercontent.com",
    "drive.google.com", "docs.google.com",
    "onedrive.live.com", "1drv.com",
    "box.com", "box.net",
    "icloud.com", "mega.nz",
    "wetransfer.com", "mediafire.com"
]);

let uploadThresholdMB = 10;

CommonSecurityLog
| where TimeGenerated > ago(1h)
| where RequestMethod in ("POST", "PUT")
| where DestinationHostName has_any (cloudStorageDomains)
| where SentBytes > uploadThresholdMB * 1024 * 1024
| extend
    CloudProvider = case(
        DestinationHostName contains "dropbox", "Dropbox",
        DestinationHostName contains "google", "Google Drive",
        DestinationHostName contains "onedrive" or DestinationHostName contains "1drv", "OneDrive Personal",
        DestinationHostName contains "box", "Box",
        DestinationHostName contains "mega", "MEGA",
        "Other Cloud Storage"
    ),
    UploadSizeMB = round(todouble(SentBytes) / (1024 * 1024), 2)
| summarize
    TotalUploadsMB = sum(UploadSizeMB),
    UploadCount = count(),
    CloudProviders = make_set(CloudProvider)
    by SourceUserName, SourceIP, bin(TimeGenerated, 1h)
| extend Severity = case(
    TotalUploadsMB > 100, "Critical",
    TotalUploadsMB > 50, "High",
    "Medium"
)
| project
    TimeGenerated,
    UserName = SourceUserName,
    SourceIP,
    TotalUploadsMB,
    UploadCount,
    CloudProviders,
    Severity,
    MITRETechnique = "T1567.002",
    RecommendedAction = "Review uploaded files, check for sensitive data"
