Cloud Security

Azure

MITRE Techniques: T1485 T1552 T1562

7.1 Mass Azure Resource Deletion

Critical

Detects bulk deletion of Azure resources within a short timeframe

T1485 
-- Rule: Mass_Azure_Resource_Delete
@RSA_Rule "Azure_Mass_Delete"
SELECT * FROM Event
WHERE 
    event_source = 'azure' AND
    operation MATCHES '*/delete' AND
    status = 'Succeeded'
GROUP BY caller_id
HAVING COUNT(*) > 10
WINDOW SLIDING 15 MINUTES
ALERT severity = 'critical',
      technique = 'T1485'
// Rule: Mass Azure Resource Deletion
// MITRE: T1485 - Data Destruction

let deletionThreshold = 10;
let timeWindow = 15m;

AzureActivity
| where TimeGenerated > ago(1h)
| where OperationNameValue endswith "/delete"
| where ActivityStatusValue == "Success"
| summarize
    DeleteCount = count(),
    ResourcesDeleted = make_set(Resource, 20),
    ResourceTypes = make_set(ResourceProviderValue),
    Subscriptions = make_set(SubscriptionId)
    by Caller, bin(TimeGenerated, timeWindow)
| where DeleteCount > deletionThreshold
| extend
    Severity = case(
        DeleteCount > 50, "Critical",
        DeleteCount > 25, "High",
        "Medium"
    )
| project
    TimeGenerated,
    Caller,
    DeleteCount,
    ResourcesDeleted,
    ResourceTypes,
    Subscriptions,
    Severity,
    MITRETechnique = "T1485",
    RecommendedAction = "Verify authorized maintenance or investigate for attack"

7.2 Sensitive Key Vault Access

Critical

Detects unusual access to Key Vault secrets from new principals

T1552.001 
-- Rule: KeyVault_Sensitive_Access
@RSA_Rule "KeyVault_Access"
SELECT * FROM Event
WHERE 
    resource_type = 'Microsoft.KeyVault' AND
    operation MATCHES '*SecretGet*|*KeyDecrypt*' AND
    caller_id NOT IN (authorized_apps)
ALERT severity = 'critical',
      technique = 'T1552.001'
// Rule: Sensitive Key Vault Access Detection
// MITRE: T1552.001 - Unsecured Credentials: Credentials In Files

let sensitiveOperations = dynamic([
    "SecretGet", "SecretList", "KeyDecrypt", 
    "KeySign", "CertificateGet"
]);

// Baseline of known accessors (30 day lookback)
let knownAccessors = AzureDiagnostics
    | where TimeGenerated between (ago(30d) .. ago(1d))
    | where ResourceType == "VAULTS"
    | where OperationName has_any (sensitiveOperations)
    | distinct CallerIPAddress, identity_claim_upn_s;

AzureDiagnostics
| where TimeGenerated > ago(1h)
| where ResourceType == "VAULTS"
| where OperationName has_any (sensitiveOperations)
| where ResultType == "Success"
| join kind=leftanti (knownAccessors) 
    on $left.identity_claim_upn_s == $right.identity_claim_upn_s
| project
    TimeGenerated,
    KeyVaultName = Resource,
    Operation = OperationName,
    CallerPrincipal = identity_claim_upn_s,
    CallerIP = CallerIPAddress,
    SecretName = tostring(split(id_s, "/")[-1]),
    Severity = "Critical",
    MITRETechnique = "T1552.001",
    RecommendedAction = "Verify access authorization, check for credential theft"

7.3 Network Security Group Modification

High

Detects NSG rule changes that expose services to the internet

T1562.007 
-- Rule: NSG_Internet_Exposure
@RSA_Rule "NSG_Modification"
SELECT * FROM Event
WHERE 
    resource_type = 'networkSecurityGroups' AND
    operation = 'write' AND
    properties MATCHES '*0.0.0.0*|*Any*'
ALERT severity = 'high',
      technique = 'T1562.007'
// Rule: NSG Internet Exposure Detection
// MITRE: T1562.007 - Impair Defenses: Disable or Modify Cloud Firewall

let riskyPorts = dynamic([22, 3389, 445, 1433, 3306, 5432]);

AzureActivity
| where TimeGenerated > ago(1h)
| where ResourceProviderValue == "MICROSOFT.NETWORK"
| where OperationNameValue has "networkSecurityGroups"
| where OperationNameValue has "write"
| where ActivityStatusValue == "Success"
| extend Props = parse_json(Properties)
| extend RequestBody = parse_json(tostring(Props.requestBody))
| where RequestBody has "0.0.0.0" or RequestBody has "*" or RequestBody has "Internet"
| extend
    NSGName = tostring(split(Resource, "/")[-1]),
    RuleDirection = iff(RequestBody has "Inbound", "Inbound", "Outbound"),
    RuleAction = iff(RequestBody has "Allow", "Allow", "Deny")
| where RuleDirection == "Inbound" and RuleAction == "Allow"
| project
    TimeGenerated,
    Caller,
    NSGName,
    SubscriptionId,
    RuleDirection,
    RuleAction,
    Severity = "High",
    MITRETechnique = "T1562.007",
    RecommendedAction = "Review NSG change, verify it doesn't expose sensitive services"

7.4 Azure RBAC Privilege Elevation

Critical

Detects assignment of Owner or Contributor roles at subscription level

T1098.001 
-- Rule: Azure_RBAC_Elevation
@RSA_Rule "RBAC_Elevation"
SELECT * FROM Event
WHERE 
    operation = 'Microsoft.Authorization/roleAssignments/write' AND
    role_name IN ('Owner', 'Contributor', 'User Access Administrator')
ALERT severity = 'critical',
      technique = 'T1098.001'
// Rule: Azure RBAC Privilege Elevation
// MITRE: T1098.001 - Account Manipulation: Additional Cloud Credentials

let highPrivilegeRoles = dynamic([
    "Owner", "Contributor", "User Access Administrator",
    "8e3af657-a8ff-443c-a75c-2fe8c4bcb635", // Owner GUID
    "b24988ac-6180-42a0-ab88-20f7382dd24c", // Contributor GUID
    "18d7d88d-d35e-4fb5-a5c3-7773c20a72d9"  // User Access Admin GUID
]);

AzureActivity
| where TimeGenerated > ago(1h)
| where OperationNameValue == "Microsoft.Authorization/roleAssignments/write"
| where ActivityStatusValue == "Success"
| extend Props = parse_json(Properties)
| extend 
    RoleDefinitionId = tostring(Props.requestbody.roleDefinitionId),
    PrincipalId = tostring(Props.requestbody.principalId),
    Scope = tostring(Props.requestbody.scope)
| where RoleDefinitionId has_any (highPrivilegeRoles)
| extend
    IsSubscriptionScope = Scope matches regex @"^/subscriptions/[^/]+$",
    RoleName = case(
        RoleDefinitionId contains "8e3af657", "Owner",
        RoleDefinitionId contains "b24988ac", "Contributor",
        RoleDefinitionId contains "18d7d88d", "User Access Administrator",
        "High Privilege Role"
    )
| project
    TimeGenerated,
    Caller,
    TargetPrincipal = PrincipalId,
    RoleName,
    Scope,
    IsSubscriptionScope,
    SubscriptionId,
    Severity = iff(IsSubscriptionScope, "Critical", "High"),
    MITRETechnique = "T1098.001",
    RecommendedAction = "Verify role assignment through change management"
