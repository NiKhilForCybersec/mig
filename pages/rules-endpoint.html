6.1 Suspicious PowerShell Execution

Critical

Detects encoded commands and suspicious PowerShell patterns

T1059.001 
-- Rule: Suspicious_PowerShell
@RSA_Rule "Suspicious_PowerShell"
SELECT * FROM Event
WHERE 
    process_name = 'powershell.exe' AND
    (command_line MATCHES '*-enc*|*-e *|*bypass*|*hidden*|*downloadstring*')
ALERT severity = 'critical',
      technique = 'T1059.001'
// Rule: Suspicious PowerShell Execution
// MITRE: T1059.001 - Command and Scripting Interpreter: PowerShell

let suspiciousPatterns = dynamic([
    "-encodedcommand", "-enc ", "-e ", "-ec ",
    "bypass", "-nop", "-w hidden", "-windowstyle hidden",
    "downloadstring", "downloadfile", "invoke-webrequest",
    "invoke-expression", "iex", "invoke-mimikatz",
    "invoke-shellcode", "new-object net.webclient"
]);

DeviceProcessEvents
| where TimeGenerated > ago(1h)
| where FileName in~ ("powershell.exe", "pwsh.exe")
| extend CmdLower = tolower(ProcessCommandLine)
| where CmdLower has_any (suspiciousPatterns)
| extend
    IsEncoded = CmdLower has_any ("-encodedcommand", "-enc ", "-e "),
    IsHidden = CmdLower has_any ("hidden", "-w h"),
    IsDownload = CmdLower has_any ("downloadstring", "downloadfile", "webclient"),
    IsBypass = CmdLower contains "bypass"
| extend RiskScore = 
    (toint(IsEncoded) * 40) + 
    (toint(IsHidden) * 20) + 
    (toint(IsDownload) * 30) + 
    (toint(IsBypass) * 10)
| project
    TimeGenerated,
    DeviceName,
    AccountName,
    ProcessCommandLine,
    InitiatingProcessFileName,
    InitiatingProcessCommandLine,
    RiskScore,
    Severity = case(
        RiskScore >= 70, "Critical",
        RiskScore >= 40, "High",
        "Medium"
    ),
    MITRETechnique = "T1059.001"

6.2 LSASS Memory Access

Critical

Detects processes accessing LSASS for credential dumping

T1003.001 
-- Rule: LSASS_Access
@RSA_Rule "LSASS_Access"
SELECT * FROM Event
WHERE 
    target_process = 'lsass.exe' AND
    access_mask IN ('0x1010', '0x1410', '0x1F1FFF') AND
    source_process NOT IN (legitimate_processes)
ALERT severity = 'critical',
      technique = 'T1003.001'
// Rule: LSASS Memory Access Detection
// MITRE: T1003.001 - OS Credential Dumping: LSASS Memory

let legitimateProcesses = dynamic([
    "csrss.exe", "wininit.exe", "smss.exe",
    "services.exe", "mrt.exe", "taskmgr.exe",
    "vmtoolsd.exe", "svchost.exe"
]);

SecurityEvent
| where TimeGenerated > ago(1h)
| where EventID == 4663
| where ObjectName endswith "lsass.exe"
| where ProcessName !in~ (legitimateProcesses)
| where AccessMask in ("0x1010", "0x1410", "0x1F1FFF", "0x143A")
| project
    TimeGenerated,
    Computer,
    Account,
    ProcessName,
    ProcessId,
    AccessMask,
    AccessList,
    Severity = "Critical",
    MITRETechnique = "T1003.001",
    RecommendedAction = "Isolate device, investigate process origin"

6.3 Process Injection Detection

High

Detects process injection using CreateRemoteThread and similar APIs

T1055 
-- Rule: Process_Injection
@RSA_Rule "Process_Injection"
SELECT * FROM Event
WHERE 
    api_call IN ('CreateRemoteThread', 'NtQueueApcThread', 'SetWindowsHookEx') AND
    source_process != target_process
ALERT severity = 'high',
      technique = 'T1055'
// Rule: Process Injection Detection
// MITRE: T1055 - Process Injection

DeviceEvents
| where TimeGenerated > ago(1h)
| where ActionType in ("CreateRemoteThreadApiCall", "QueueUserApcRemoteApiCall")
| where InitiatingProcessId != ProcessId
| extend
    InjectionType = case(
        ActionType == "CreateRemoteThreadApiCall", "CreateRemoteThread",
        ActionType == "QueueUserApcRemoteApiCall", "APC Injection",
        "Unknown"
    )
| project
    TimeGenerated,
    DeviceName,
    InitiatingProcessFileName,
    InitiatingProcessCommandLine,
    FileName,
    ProcessCommandLine,
    InjectionType,
    Severity = "High",
    MITRETechnique = "T1055"
