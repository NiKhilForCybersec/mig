SalesforceParser

Parses Salesforce Shield Event Monitoring logs and normalizes fields for detection rules.

// Function: SalesforceParser
// Purpose: Parse and normalize Salesforce Shield Event Monitoring logs
// Usage: SalesforceParser(ago(1h), now())

let SalesforceParser = (startTime: datetime, endTime: datetime) {
    Salesforce_CL
    | where TimeGenerated between (startTime .. endTime)
    | extend EventData = parse_json(RawData)
    | extend
        // Core event fields
        SF_Timestamp = todatetime(EventData.TIMESTAMP_DERIVED),
        SF_EventType = tostring(EventData.EVENT_TYPE),
        SF_UserId = tostring(EventData.USER_ID),
        SF_UserName = tostring(EventData.USER_NAME),
        SF_UserType = tostring(EventData.USER_TYPE),
        
        // Session and client info
        SF_SessionId = tostring(EventData.SESSION_KEY),
        SF_ClientIP = tostring(EventData.SOURCE_IP),
        SF_LoginGeo = EventData.LOGIN_GEO_ID,
        SF_Browser = tostring(EventData.BROWSER_TYPE),
        SF_Platform = tostring(EventData.PLATFORM_TYPE),
        
        // API and query fields
        SF_ApiType = tostring(EventData.API_TYPE),
        SF_ApiVersion = tostring(EventData.API_VERSION),
        SF_HttpMethod = tostring(EventData.HTTP_METHOD),
        SF_Uri = tostring(EventData.URI),
        SF_Query = tostring(EventData.QUERY),
        
        // Entity and data fields
        SF_EntityName = tostring(EventData.ENTITY_NAME),
        SF_RecordId = tostring(EventData.RECORD_ID),
        SF_RowsProcessed = toint(EventData.ROWS_PROCESSED),
        SF_RowCount = toint(EventData.ROW_COUNT),
        
        // Report and export fields
        SF_ReportId = tostring(EventData.REPORT_ID),
        SF_ReportName = tostring(EventData.REPORT_NAME),
        SF_ExportType = tostring(EventData.EXPORT_TYPE),
        
        // Login specific
        SF_LoginStatus = tostring(EventData.LOGIN_STATUS),
        SF_LoginType = tostring(EventData.LOGIN_TYPE),
        
        // Calculated fields
        SF_RecordsProcessed = coalesce(toint(EventData.ROWS_PROCESSED), toint(EventData.ROW_COUNT), 0)
    | project-away RawData
};

// Function: CEFEnhancedParser
// Purpose: Enhanced CEF parsing with vendor normalization

let CEFEnhancedParser = (startTime: datetime, endTime: datetime) {
    CommonSecurityLog
    | where TimeGenerated between (startTime .. endTime)
    | extend
        // Normalize action across vendors
        NormalizedAction = case(
            DeviceAction in ("allow", "permitted", "pass", "accept"), "Allow",
            DeviceAction in ("deny", "blocked", "drop", "reject"), "Block",
            DeviceAction in ("reset", "reset-both", "rst"), "Reset",
            "Other"
        ),
        
        // Normalize severity
        NormalizedSeverity = case(
            LogSeverity >= 9, "Critical",
            LogSeverity >= 7, "High",
            LogSeverity >= 4, "Medium",
            LogSeverity >= 1, "Low",
            "Informational"
        ),
        
        // Extract vendor-specific threat categories
        ThreatCategory = case(
            DeviceVendor == "Palo Alto Networks", tostring(DeviceCustomString1),
            DeviceVendor == "Fortinet", tostring(DeviceCustomString2),
            DeviceVendor == "Cisco", Activity,
            "Unknown"
        ),
        
        // Calculate bytes transferred
        TotalBytes = SentBytes + ReceivedBytes,
        
        // Normalize direction
        TrafficDirection = case(
            CommunicationDirection == "1" or CommunicationDirection == "inbound", "Inbound",
            CommunicationDirection == "0" or CommunicationDirection == "outbound", "Outbound",
            "Unknown"
        )
    // Join with Threat Intelligence
    | join kind=leftouter (
        ThreatIntelligenceIndicator
        | where isnotempty(NetworkIP)
        | summarize arg_max(TimeGenerated, *) by NetworkIP
        | project TI_IP = NetworkIP, TI_Type = ThreatType, TI_Confidence = ConfidenceScore
    ) on $left.SourceIP == $right.TI_IP
    | extend IsThreatIntelMatch = isnotempty(TI_IP)
};

WindowsSecurityParser

ASIM-compatible parser for Windows Security Events with logon type decoding and privilege tracking.

// Function: WindowsSecurityParser
// Purpose: ASIM-compatible Windows Security Event parser

let WindowsSecurityParser = (startTime: datetime, endTime: datetime) {
    SecurityEvent
    | where TimeGenerated between (startTime .. endTime)
    | where EventID in (4624, 4625, 4634, 4648, 4672, 4688, 4720, 4728, 4732)
    | extend
        // Decode logon type
        LogonTypeName = case(
            LogonType == 2, "Interactive",
            LogonType == 3, "Network",
            LogonType == 4, "Batch",
            LogonType == 5, "Service",
            LogonType == 7, "Unlock",
            LogonType == 8, "NetworkCleartext",
            LogonType == 9, "NewCredentials",
            LogonType == 10, "RemoteInteractive",
            LogonType == 11, "CachedInteractive",
            "Unknown"
        ),
        
        // Determine event category
        EventCategory = case(
            EventID in (4624, 4625, 4634, 4648), "Authentication",
            EventID == 4672, "PrivilegeUse",
            EventID == 4688, "ProcessCreation",
            EventID in (4720, 4728, 4732), "AccountManagement",
            "Other"
        ),
        
        // Identify privileged logons
        IsPrivilegedLogon = EventID == 4672,
        
        // Check for service accounts
        IsServiceAccount = TargetUserName matches regex "^(svc_|srv_|service_|SA_)",
        
        // Parse sensitive privileges
        HasSensitivePrivileges = PrivilegeList has_any (
            "SeDebugPrivilege",
            "SeTcbPrivilege",
            "SeLoadDriverPrivilege",
            "SeBackupPrivilege",
            "SeRestorePrivilege"
        ),
        
        // ASIM schema mapping
        EventType = case(
            EventID == 4624, "Logon",
            EventID == 4625, "Logon",
            EventID == 4634, "Logoff",
            "Other"
        ),
        EventResult = iff(EventID == 4625, "Failure", "Success"),
        DvcHostname = Computer,
        ActorUsername = strcat(TargetDomainName, "\\", TargetUserName),
        SrcIpAddr = IpAddress
};

// Function: AzureActivityParser
// Purpose: Parse Azure Activity logs with resource hierarchy

let AzureActivityParser = (startTime: datetime, endTime: datetime) {
    AzureActivity
    | where TimeGenerated between (startTime .. endTime)
    | extend
        // Parse resource hierarchy
        ResourceParts = split(_ResourceId, "/"),
        SubscriptionId = tostring(split(_ResourceId, "/")[2]),
        ResourceGroup = tostring(split(_ResourceId, "/")[4]),
        ResourceProvider = tostring(split(_ResourceId, "/")[6]),
        ResourceType = tostring(split(_ResourceId, "/")[7]),
        ResourceName = tostring(split(_ResourceId, "/")[8]),
        
        // Identify sensitive operations
        IsSensitiveOperation = OperationNameValue has_any (
            "Microsoft.Authorization/roleAssignments/write",
            "Microsoft.Authorization/roleDefinitions/write",
            "Microsoft.KeyVault/vaults/secrets",
            "Microsoft.Compute/virtualMachines/delete",
            "Microsoft.Storage/storageAccounts/delete",
            "Microsoft.Network/networkSecurityGroups/securityRules/write"
        ),
        
        // Categorize operation type
        OperationType = case(
            OperationNameValue endswith "/write", "Create/Update",
            OperationNameValue endswith "/delete", "Delete",
            OperationNameValue endswith "/action", "Action",
            OperationNameValue endswith "/read", "Read",
            "Other"
        ),
        
        // Detect RBAC changes
        IsRBACChange = OperationNameValue startswith "Microsoft.Authorization/roleAssignments",
        
        // Detect policy changes
        IsPolicyChange = OperationNameValue startswith "Microsoft.Authorization/policyAssignments",
        
        // Parse caller identity
        CallerType = case(
            Caller contains "@", "User",
            Caller matches regex "^[0-9a-f-]{36}$", "ServicePrincipal",
            "Unknown"
        ),
        
        // Determine result
        OperationResult = iff(ActivityStatusValue == "Success", "Success", "Failure")
    | project-reorder
        TimeGenerated, Caller, CallerType, OperationNameValue, 
        OperationType, OperationResult, ResourceGroup, ResourceName
};

// Detect mass data exports from Salesforce
SalesforceParser(ago(1h), now())

| where SF_RecordsProcessed > 10000
| project SF_Timestamp, SF_UserName, SF_EventType, SF_RecordsProcessed

// Find privileged logons by service accounts
WindowsSecurityParser(ago(24h), now())
| where IsPrivilegedLogon and IsServiceAccount
| where HasSensitivePrivileges
| summarize PrivilegedLogons = count() by ActorUsername, DvcHostname
