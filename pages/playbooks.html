# RSA Playbook: Disable_Compromised_Account
name: Disable_Compromised_Account
trigger:
  alert_type: "Account_Compromise"
  severity: ["critical", "high"]

actions:
  - name: lookup_user
    type: active_directory_lookup
    parameters:
      username: "${alert.user_name}"
      
  - name: check_vip
    type: condition
    condition: "${lookup_user.department} NOT IN ['Executive', 'C-Suite']"
    
  - name: disable_account
    type: active_directory_action
    action: disable_user
    parameters:
      username: "${alert.user_name}"
    when: "${check_vip} == true"
    
  - name: notify_manager
    type: email
    parameters:
      to: "${lookup_user.manager_email}"
      subject: "Account Disabled: ${alert.user_name}"
      body: "Account disabled due to security incident"
      
  - name: create_ticket
    type: servicenow_create
    parameters:
      category: "Security Incident"
      priority: "P1"
      description: "${alert.description}"
      
  - name: isolate_endpoint
    type: edr_action
    action: isolate
    parameters:
      hostname: "${alert.source_host}"
    when: "${alert.severity} == 'critical'"

RSA Actions

  1. Lookup user in Active Directory
  2. Check if VIP user
  3. Disable account (if not VIP)
  4. Send email notification to manager
  5. Create ServiceNow ticket
  6. Isolate endpoint (if critical)

Logic App Actions

  1. Parse incident entities from Sentinel
  2. Query user details from Entra ID
  3. Check department for VIP status
  4. Disable account + revoke sessions
  5. Get manager and send notification
  6. Create ServiceNow incident
  7. Isolate device via MDE API
  8. Update Sentinel incident with comments

  alert_type: "Salesforce_Mass_Export"
  severity: ["critical"]

  - name: freeze_sf_user
    type: salesforce_api
    action: freeze_user
    parameters:
      user_id: "${alert.sf_user_id}"
