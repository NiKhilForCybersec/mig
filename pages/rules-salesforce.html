Salesforce Security

Critical

24 RSA rules migrated to 28 Sentinel analytics rules with custom Salesforce Shield Event Monitoring parser for comprehensive CRM security.

MITRE Techniques: T1567 T1098 T1078

Prerequisites

These rules require Salesforce Shield Event Monitoring connector and the custom SalesforceParser KQL function.

4.1 Salesforce Mass Data Export

Critical

Detects export of large datasets (10,000+ records) from Salesforce

T1567 
-- Rule: Salesforce_Mass_Export
@RSA_Rule "SF_Mass_Export"
SELECT * FROM Event
WHERE 
    event_source = 'salesforce' AND
    event_type IN ('ReportExport', 'DataExport', 'ApiQuery') AND
    records_exported > 10000
ALERT severity = 'critical',
      technique = 'T1567'
// Rule: Salesforce Mass Data Export Detection
// MITRE: T1567 - Exfiltration Over Web Service

let exportThreshold = 10000;

SalesforceParser(ago(1h), now())
| where SF_EventType in ("ReportExport", "DataExport", "BulkApi")
| where SF_RowsProcessed > exportThreshold or SF_RowCount > exportThreshold
| extend RecordsExported = coalesce(SF_RowsProcessed, SF_RowCount)
| project
    TimeGenerated = SF_Timestamp,
    UserName = SF_UserName,
    ClientIP = SF_ClientIP,
    EventType = SF_EventType,
    EntityName = SF_EntityName,
    RecordsExported,
    Severity = case(
        RecordsExported > 100000, "Critical",
        RecordsExported > 50000, "High",
        "Medium"
    ),
    MITRETechnique = "T1567"

4.2 Salesforce Permission Escalation

Critical

Detects assignment of high-privilege permission sets

T1098 
-- Rule: Salesforce_Permission_Escalation
@RSA_Rule "SF_Permission_Escalation"
SELECT * FROM Event
WHERE 
    event_source = 'salesforce' AND
    event_type = 'PermissionSetAssignment' AND
    permission_set IN ('SystemAdministrator', 'ModifyAllData', 'ViewAllData')
ALERT severity = 'high',
      technique = 'T1098'
// Rule: Salesforce Permission Escalation
// MITRE: T1098 - Account Manipulation

let highPrivilegePermissions = dynamic([
    'SystemAdministrator', 'ModifyAllData', 'ViewAllData', 
    'ManageUsers', 'ManageProfiles', 'ApiEnabled'
]);

SalesforceParser(ago(1d), now())
| where SF_EventType == "PermissionSetAssignment"
| where SF_PermissionSet has_any (highPrivilegePermissions)
| project
    TimeGenerated = SF_Timestamp,
    AssigningUser = SF_UserName,
    TargetUser = SF_DelegateUser,
    PermissionAssigned = SF_PermissionSet,
    ClientIP = SF_ClientIP,
    Severity = "High",
    MITRETechnique = "T1098"

4.3 Salesforce Session Hijacking

Critical

Detects same session used from multiple IP addresses

T1563 
-- Rule: Salesforce_Session_Hijack
@RSA_Rule "SF_Session_Hijack"
SELECT * FROM Event
WHERE event_source = 'salesforce' AND session_id IS NOT NULL
GROUP BY session_id
HAVING COUNT(DISTINCT source_ip) > 1
WINDOW SLIDING 30 MINUTES
ALERT severity = 'critical', technique = 'T1563'
// Rule: Salesforce Session Hijacking Detection
// MITRE: T1563 - Remote Service Session Hijacking

SalesforceParser(ago(30m), now())
| where isnotempty(SF_SessionId)
| summarize
    UniqueIPs = dcount(SF_ClientIP),
    IPList = make_set(SF_ClientIP),
    EventCount = count()
    by SF_SessionId, SF_UserName
| where UniqueIPs > 1
| project
    TimeGenerated = now(),
    SessionId = SF_SessionId,
    UserName = SF_UserName,
    UniqueIPs,
    IPList,
    Severity = "Critical",
    MITRETechnique = "T1563",
    RecommendedAction = "Terminate session immediately"
